46 // Defines a matcher that matches an empty container. The container must
47 // support both size() and empty(), which all STL-like containers provide.
48 MATCHER(IsEmpty, negation ? "isn't empty" : "is empty") {
49  if (arg.empty()) {
50  return true;
51  }
52  *result_listener << "whose size is " << arg.size();
53  return false;
54 }

56 // Define a matcher that matches a value that evaluates in boolean
57 // context to true. Useful for types that define "explicit operator
58 // bool" operators and so can't be compared for equality with true
59 // and false.
60 MATCHER(IsTrue, negation ? "is false" : "is true") {
61  return static_cast<bool>(arg);
62 }
63 
64 // Define a matcher that matches a value that evaluates in boolean
65 // context to false. Useful for types that define "explicit operator
66 // bool" operators and so can't be compared for equality with true
67 // and false.
68 MATCHER(IsFalse, negation ? "is true" : "is false") {
69  return !static_cast<bool>(arg);
