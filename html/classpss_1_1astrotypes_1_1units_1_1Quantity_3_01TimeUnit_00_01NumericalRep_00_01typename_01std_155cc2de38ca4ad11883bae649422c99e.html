template<typename TimeUnit, typename NumericalRep>
class pss::astrotypes::units::Quantity< TimeUnit, NumericalRep, typename std::enable_if< boost::units::is_unit_of_dimension< TimeUnit, boost::units::time_dimension >::value >::type >

Specialisation of Quantity unit for time dimension types.

all the functionality of the normal Quantity object plus extra functions to ensure interoperability with std::chrono types
