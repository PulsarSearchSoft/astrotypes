510 bool Slice<is_const, Parent, SliceMixin, Dimension, Dimensions...>::increment_it(IteratorT& current, SlicePosition<rank>& pos) const
511 {
512  if(!BaseT::increment_it(current, pos)) {
513  current -= BaseT::diff_base_span(); // return pointer to beginning of BaseT block
514  if(++pos.index < _span.span())
515  {
516  current += BaseT::_base_span; // move up to next block

521  return false;
522  }
